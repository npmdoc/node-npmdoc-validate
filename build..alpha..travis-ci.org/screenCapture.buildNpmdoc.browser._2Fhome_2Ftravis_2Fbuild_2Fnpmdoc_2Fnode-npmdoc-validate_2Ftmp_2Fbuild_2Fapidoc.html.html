function validate.property.prototype.validate (value, ctx)

  • description and source-code
    validate = function (value, ctx) {
  var fns = this.fns;

  for (var i = 0; i < fns.length; i++) {
    var fn = fns[i];
    var valid = fn[0].call(ctx, value);
    if (!valid) return error(this, fn[1]);
  }

  return false;
}
